  from langgraph.store.sqlite.aio import AsyncSqliteStore  # 新增

  from agent.config import settings

  from agent.models.llm import llm
  from agent.tools.internet_search import internet_search

  _graph = None
  _store = None  # 新增：模块级 store 引用，供 API 层访问


  def get_graph():
      if _graph is None:
          raise RuntimeError("Graph not initialized. Call init_graph() first.")
      return _graph


  def get_store():
      """Return the initialized store (for memory management API)."""
      if _store is None:
          raise RuntimeError("Store not initialized. Call init_graph() first.")
      return _store


  async def init_graph():
      global _graph, _store

      # Checkpointer (已有)
      checkpoint_conn = await aiosqlite.connect(settings.CHECKPOINT_DB_PATH)
      checkpointer = AsyncSqliteSaver(checkpoint_conn)

      # Store (新增)
      store_conn = await aiosqlite.connect(settings.STORE_DB_PATH)
      _store = AsyncSqliteStore(store_conn)
      await _store.setup()  # 创建 store 相关表

      # 用户隔离的 namespace 函数
      def _user_namespace(rt):
          return (rt.context.user_id,)

      _graph = create_deep_agent(
          model=llm,
          tools=[internet_search],
          checkpointer=checkpointer,
          store=_store,                                              # 新增
          memory=["/memories/preferences.md"],                       # 
  新增：记忆文件列表
          backend=lambda rt: CompositeBackend(
              default=StateBackend(rt),
              routes={
                  "/memories/": StoreBackend(rt, namespace=_user_namespace),
              },
          ),                                                         # 
  新增：路由 /memories/ 到持久化 store
          context_schema=AgentContext,                               # 
  新增：声明 context 类型
          system_prompt="你是 ke-hermes 
  通用智能体，请根据用户的需求提供准确、有用的回答。",
      )

  核心设计点：
  - CompositeBackend 把 /memories/ 路径路由到 StoreBackend，其余文件走默认的
  StateBackend（线程内临时）
  - StoreBackend 的 namespace=_user_namespace 保证了不同用户的记忆文件互不可见
  - memory=["/memories/preferences.md"] 告诉 agent
  启动时自动加载偏好文件到上下文中

  步骤 5：添加 JWT 鉴权依赖

  新建文件：backend/src/api/deps_auth.py

  from fastapi import Depends, Header, HTTPException, status

  from core.security import decode_token


  async def get_current_user_id(
      authorization: str = Header(..., description="Bearer {access_token}"),
  ) -> str:
      """Extract user_id from JWT Bearer token. Returns 401 if invalid."""
      if not authorization.startswith("Bearer "):
          raise HTTPException(
              status_code=status.HTTP_401_UNAUTHORIZED,
              detail="Missing or invalid Authorization header",
          )
      token = authorization.removeprefix("Bearer ").strip()
      payload = decode_token(token)
      if payload is None:
          raise HTTPException(
              status_code=status.HTTP_401_UNAUTHORIZED,
              detail="Invalid or expired token",
          )
      return payload["sub"]

  这里复用了现有的 decode_token（core/security.py），它已经能验证 JWT 并返回
  payload。sub 字段存储的是 user_id（UUID 字符串）。

  修改文件：backend/src/api/agent/agent_api.py

  import json

  from fastapi import APIRouter, Depends
  from fastapi.responses import StreamingResponse
  from langchain_core.messages import HumanMessage
  from langchain_core.utils.uuid import uuid7
  from pydantic import BaseModel, Field

  from agent import get_graph
  from agent.context import AgentContext
  from api.deps_auth import get_current_user_id

  router = APIRouter(prefix="/api")


  class ChatRequest(BaseModel):
      message: str = Field(min_length=1)
      thread_id: str | None = None


  class ChatResponse(BaseModel):
      response: str
      thread_id: str


  @router.post("/chat", response_model=ChatResponse)
  async def chat(
      req: ChatRequest,
      user_id: str = Depends(get_current_user_id),  # 新增：JWT 鉴权
  ):
      thread_id = req.thread_id or str(uuid7())
      config = {"configurable": {"thread_id": thread_id}}
      context = AgentContext(user_id=user_id)        # 新增：构建上下文

      result = await get_graph().ainvoke(
          {"messages": [HumanMessage(content=req.message)]},
          config=config,
          context=context,                            # 新增：传入 context
      )
      final_message = result["messages"][-1]
      return ChatResponse(response=final_message.content, thread_id=thread_id)


  @router.post("/chat/stream")
  async def chat_stream(
      req: ChatRequest,
      user_id: str = Depends(get_current_user_id),  # 新增
  ):
      thread_id = req.thread_id or str(uuid7())
      config = {"configurable": {"thread_id": thread_id}}
      context = AgentContext(user_id=user_id)

      async def event_generator():
          async for event in get_graph().astream_events(
              {"messages": [HumanMessage(content=req.message)]},
              config=config,
              context=context,
              version="v2",
          ):
              if event["event"] == "on_chat_model_stream":
                  token = event["data"]["chunk"].text
                  if token:
                      yield f"data: {json.dumps({'token': token})}\n\n"
          yield f"data: {json.dumps({'thread_id': thread_id})}\n\n"

      return StreamingResponse(event_generator(),
  media_type="text/event-stream")

  变更要点：
  - 两个 chat 端点都加了 user_id: str = Depends(get_current_user_id) →
  未登录调用直接 401
  - graph.ainvoke() / graph.astream_events() 都传入
  context=AgentContext(user_id=user_id)，使 StoreBackend 的 namespace
  函数能获取到 user_id

  class ThreadCreate(BaseModel):
      thread_id: str
      title: str = "新对话"


  class ThreadUpdate(BaseModel):
      title: str | None = None


  class ThreadResponse(BaseModel):
      id: str
      title: str
      created_at: str
      updated_at: str


  # ---- 列表 ----
  @router.get("/chat/threads", response_model=list[ThreadResponse])
  async def list_threads(
      user_id: str = Depends(get_current_user_id),
      db: AsyncSession = Depends(get_db),
  ):
      result = await db.execute(
          select(ChatThread)
          .where(ChatThread.user_id == user_id)
          .order_by(desc(ChatThread.updated_at))
      )
      threads = result.scalars().all()
      return [
          ThreadResponse(
              id=t.id,
              title=t.title,
              created_at=t.created_at.isoformat(),
              updated_at=t.updated_at.isoformat(),
          )
          for t in threads
      ]

  async def create_thread(
      req: ThreadCreate,
      user_id: str = Depends(get_current_user_id),
      db: AsyncSession = Depends(get_db),
  ):
      thread = ChatThread(id=req.thread_id, user_id=user_id, title=req.title)
      db.add(thread)
      await db.flush()
      return {"id": thread.id, "title": thread.title}


  # ---- 更新标题 ----
  @router.patch("/chat/threads/{thread_id}")
  async def update_thread(

      )
      thread = result.scalar_one_or_none()
      if not thread:
          raise HTTPException(status_code=404, detail="Thread not found")
      if req.title is not None:
          thread.title = req.title
      await db.flush()
      return {"id": thread.id, "title": thread.title}

  import { defineStore } from 'pinia'
  import { ref } from 'vue'
  import {
    fetchThreads,
    createThread,
    updateThread,
    deleteThread,
    type ThreadItem,
  } from '@/services/chatHistoryApi'

  export interface HistoryItem {
    id: number           // 本地 UI 用的自增 ID（保留用于 key）
    threadId: string      // 新增：服务端 thread_id
    title: string
  }

  export const useUiStore = defineStore('ui', () => {
    const sidebarCollapsed = ref(false)
    const rightPanelCollapsed = ref(false)
    const plusMenuOpen = ref(false)
    const searchQuery = ref('')
    const selectedModel = ref('DeepSeek V4')

    // 废弃硬编码数据，改为从 API 加载
    const histories = ref<HistoryItem[]>([])
    const activeThreadId = ref<string | null>(null)  // 改为 threadId
    let nextLocalId = 1

    // 从服务端加载对话历史
    async function loadThreads() {
      try {
        const threads = await fetchThreads()
        histories.value = threads.map((t: ThreadItem) => ({
          id: nextLocalId++,
          threadId: t.id,
          title: t.title,
        }))
      } catch {
        // 未登录或网络错误时忽略，保持空列表
      }
    }

    // 创建新对话（同步到服务端）
    async function createNewThread(threadId: string, title = '新对话') {
      try {
        await createThread(threadId, title)
      } catch {
        // 即使服务端失败，本地仍然记录
      }
      const item: HistoryItem = {
        id: nextLocalId++,
        threadId,
        title,
      }
      histories.value.unshift(item)
      activeThreadId.value = threadId
      plusMenuOpen.value = false
      return threadId
    }

    // 重命名对话
    async function renameThread(threadId: string, title: string) {
      const h = histories.value.find(h => h.threadId === threadId)
      if (h) h.title = title
      try {
        await updateThread(threadId, title)
      } catch { /* ignore */ }
    }

    // 删除对话
    async function removeThread(threadId: string) {
      histories.value = histories.value.filter(h => h.threadId !== threadId)
      if (activeThreadId.value === threadId) {
        activeThreadId.value = null
      }
      try {
        await deleteThread(threadId)
      } catch { /* ignore */ }
    }

    function toggleSidebar() { sidebarCollapsed.value = !sidebarCollapsed.value
  }
    function toggleRightPanel() { rightPanelCollapsed.value =
  !rightPanelCollapsed.value }
    function togglePlusMenu() { plusMenuOpen.value = !plusMenuOpen.value }
    function closePlusMenu() { plusMenuOpen.value = false }

    return {
      sidebarCollapsed, rightPanelCollapsed, plusMenuOpen,
      searchQuery, selectedModel,
      histories, activeThreadId,
      loadThreads, createNewThread, renameThread, removeThread,
      toggleSidebar, toggleRightPanel, togglePlusMenu, closePlusMenu,
    }
  })

  import { generateThreadTitle } from '@/utils/threadTitle'  // 工具函数

  // ... ChatMessage 接口不变 ...

  export const useChatStore = defineStore('chat', () => {
    const messages = ref<ChatMessage[]>([])
    const loading = ref(false)
    const threadId = ref<string | null>(null)
    let nextId = 1

    async function sendMessage(text: string) {
      if (loading.value || !text.trim()) return

      loading.value = true
      const isNewThread = !threadId.value

      // 如果是新对话，先生成 threadId
      if (isNewThread) {
        threadId.value = crypto.randomUUID()
      }

      addMessage('user', text.trim())
      const assistantId = addMessage('assistant', '', true).id

      // 收集 assistant 完整回复用于自动生成标题
      let fullResponse = ''

      try {
        await sendStreamRequest(text.trim(), {
          threadId: threadId.value,
          onToken(token: string) {
            fullResponse += token
            const idx = messages.value.findIndex(m => m.id === assistantId)
            if (idx !== -1) {
              messages.value[idx] = { ...messages.value[idx], content:
  messages.value[idx].content + token }
            }
          },
          onThreadId(id: string) {
            threadId.value = id
          },
          onDone() {
            const idx = messages.value.findIndex(m => m.id === assistantId)
            if (idx !== -1) {
              messages.value[idx] = { ...messages.value[idx], streaming: false }
            }
            loading.value = false

            // 新对话首轮完成后：同步到服务端历史 + 自动生成标题
            if (isNewThread && threadId.value) {
              const ui = useUiStore()
              const title = generateThreadTitle(fullResponse)
              ui.createNewThread(threadId.value, title)
            }
          },
          onError(err: Error) { /* ... 同现有逻辑 ... */ },
        })
      } catch (err) { /* ... 同现有逻辑 ... */ }
    }

    // 切换到已有对话（从历史列表点击）
    async function switchToThread(newThreadId: string) {
      // 注意：此操作需要从 checkpointer 恢复历史消息
      // MVP 方案：清空消息列表，让 agent 从 checkpoint 自动恢复
      // 前端不加载历史消息，后端 astream_events 会自动恢复
      threadId.value = newThreadId
      messages.value = []
      nextId = 1
    }

    function clearMessages() {
      messages.value = []
      loading.value = false
      threadId.value = null
    }

    return { messages, loading, threadId, sendMessage, switchToThread,
  clearMessages }
  })

  export function generateThreadTitle(response: string): string {
    const text = response.replace(/[#*`>\-\[\]()]/g, '').trim()
    const firstLine = text.split('\n')[0]
    return firstLine.slice(0, 30) + (firstLine.length > 30 ? '...' : '')
  }

  import { ref } from 'vue'
  import { fetchMemoryFiles, deleteMemoryFile } from '@/services/chatHistoryApi'

  export function useMemory() {
    const memoryFiles = ref<Array<{ key: string; value: string; updated_at:
  string | null }>>([])
    const loading = ref(false)

    async function load() {
      loading.value = true
      try {
        memoryFiles.value = await fetchMemoryFiles()
      } catch {
        memoryFiles.value = []
      } finally {
        loading.value = false
      }
    }

    async function remove(key: string) {
      try {
        await deleteMemoryFile(key)
        memoryFiles.value = memoryFiles.value.filter(f => f.key !== key)
      } catch { /* ignore */ }
    }

    return { memoryFiles, loading, load, remove }
  }
